Import clerkMiddleware from @clerk/nextjs and apply it in the middleware function. Clerk Core 3 uses `clerkMiddleware()` (not `authMiddleware()`).

Add import at the top:
```typescript
import { clerkMiddleware } from '@clerk/nextjs/server'
```

Replace the existing middleware function body with clerkMiddleware. The clerkMiddleware function returns a NextResponse (or redirect) directly. The pattern for Clerk Core 3 is:

```typescript
// Remove: const { pathname } = request.nextUrl, rate limiting, CORS logic from before Clerk

export default clerkMiddleware((auth, request) => {
  // auth().protect() marks routes that require authentication
  // After Clerk, you can still keep CORS + rate limiting by composing with NextResponse

  const { pathname } = request.nextUrl

  // CORS for preflight OPTIONS requests
  if (request.method === 'OPTIONS') {
    const origin = request.headers.get('origin')
    const allowedOrigins = [
      process.env.NEXT_PUBLIC_SITE_URL || 'https://greenlog.app',
      'http://localhost:3000',
    ]
    if (origin && allowedOrigins.includes(origin)) {
      return new NextResponse(null, {
        status: 204,
        headers: {
          'Access-Control-Allow-Origin': origin,
          'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, PATCH, OPTIONS',
          'Access-Control-Allow-Headers': 'Content-Type, Authorization, X-Requested-With',
          'Access-Control-Max-Age': '86400',
        },
      })
    }
    return NextResponse.json({}, { status: 405 })
  }

  // Rate limiting (reuse existing Upstash Redis logic)
  const ip = request.headers.get('x-forwarded-for') || 'anonymous'
  const isLimited = await isRateLimited(ip, pathname)
  if (isLimited) {
    const isApi = pathname.startsWith('/api')
    const retryAfter = isApi ? 60 : 60
    return NextResponse.json(
      { error: 'Too many requests', retryAfter },
      { status: 429, headers: { 'Retry-After': String(retryAfter) } }
    )
  }

  // Allow Clerk auth to proceed
  return NextResponse.next()
})
```

Keep the existing `getRatelimits()`, `isRateLimited()`, `getAllowedOrigins()` helper functions and the `export const config` at the bottom unchanged. Only replace the `export async function middleware` implementation.
